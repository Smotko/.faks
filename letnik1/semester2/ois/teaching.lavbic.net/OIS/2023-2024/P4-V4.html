Poglavje 8 P4 → V4 Odjemalec/strežnik in Node.js

P4 → V4

V pripravah na vaje V4 bo predstavljen kratek uvod v Node.js:

  • enostavni spletni strežnik (Hello World; testiranje in ponovni zagon; spremenljivke na strežniku),
  • napredni spletni strežnik (strežnik statičnih spletnih strani; uporaba dodatnih knjižnic; obvladovanje zahtev po virih, ki ne obstajajo; predpomnjenje; dinamične strani; pridobivanje podatkov uporabnika; shranjevanje v datoteko).

Celotna izvorna koda primera v nadaljevanju je na voljo v GitHub repozitoriju P4-V4. Najprej na GitHub izdelajte kopijo projekta (angl. Fork), nato pa si lokalno okolje nastavite z naslednjima ukazoma:

git clone git@github.com:{VAŠE-GITHUB-UPORABNIŠKO-IME}/P4-V4.git
cd P4-V4

ali

git clone https://github.com/{VAŠE-GITHUB-UPORABNIŠKO-IME}/P4-V4.git
cd P4-V4

Pri delu bomo uporabljali razvojno okolje Node.js, ki ga morate namestiti na svoj računalnik. Node.js je odprtokodno ogrodje, ki omogoča izvajanje kode v programskem jeziku JavaScript na strani strežnika in deluje na vseh pogosto uporabljenih platformah (Windows, Linux in macOS).

8.1 Razvoj enostavnega spletnega strežnika

Spremembe v repozitoriju od tega koraka naprej so na voljo na uveljavitvi s sporočilom Enostavni strežnik.

8.1.1 Hello World

Kreiramo najbolj enostavno spletno aplikacijo, ki izpiše zgolj pozdrav. Uporabimo vgrajeno knjižnico http in metodi http.createServer za kreiranje objekta strežnika ter server.listen za zagon strežnika na izbranih vratih (angl. port).

streznik_enostavni.js

/**
 * Če so vrata strežnika opredeljena v okoljski spremenljivki PORT,
 * uporabimo to vrednost, sicer nastavimo vrata na vrednost 8080.
 */
const port = process.env.PORT || 8080;

const http = require("http");
http
  .createServer(function (zahteva, odgovor) {
    odgovor.writeHead(200, { "Content-Type": "text/plain" });
    odgovor.end("Pozdravljeni vsi ljubitelji predmeta OIS!\n");
  })
  .listen(port);

console.log("Strežnik je pognan!");

Tukaj se zopet (prvič smo ga omenili na prejšnjih pripravah na vaje P3 → V3 v okviru poglavja Dogodki in povratni klici) srečamo s funkcionalnostjo povratnega klica (angl. callback), ki je zelo pogost vzorec pri JavaScript asinhronih operacijah.

Funkcija povratnega klica je funkcija (npr. funkcijaX), ki jo posredujemo kot parameter pri klicu druge funkcije (npr. funkcijaY) in ta funkcija povratnega klica (funkcijaX) se izvede znotraj druge funkcije (funkcijaY). Več o tej funkcionalnosti si lahko preberete v prispevku Understand JavaScript Callback Functions and Use Them.

8.1.1.1 Zagon strežnika in pregled delovanja

Strežniško aplikacijo, ki smo jo ravnokar razvili, lahko v lokalnem razvojnem okolju poženemo z uporabo ukaza node streznik_enostavni.js, ki ga vnesemo v terminal.

8.1.1.2 Spremembe in ponovni zagon

Če spremenimo izpis (glej spodaj), nam strežnika ni potrebno ponovno pognati, v brskalniku samo osvežimo stran in imamo na voljo novo vsebino!

streznik_enostavni.js

http
  .createServer(function (zahteva, odgovor) {
    odgovor.writeHead(200, { "Content-Type": "text/plain" });
    odgovor.end(
      "Pozdravljeni vsi ljubitelji predmeta OIS in ostali slepi potniki!\n"
    );
  })
  .listen(port);

Če so spremembe na strani strežnika večje, potem moramo strežnik ponovno pognati (v terminalu trenutno aktivni strežnik ustavimo (npr. s kombinacijo tipk CTRL / CMD + C in ga ponovno poženemo s prej omenjenim ukazom node enostavni_streznik.js).

8.2 Spremenljivke na strežniku

Dodajmo enostaven števec uporabnikov, ki hrani število uporabnikov v globalni spremenljivki na strežniku.

streznik_enostavni.js

const http = require("http");
let steviloUporabnikov = 0;
http
  .createServer(function (zahteva, odgovor) {

8.3.6 Pripravimo še manjkajočo metodo za strežbo statične vsebine

Uporabili bomo metodo fs.access, ki preveri, ali datoteka na datotečnem sistemu obstaja, in jo bomo v primeru obstoja prebrali z metodo fs.readFile.

streznik_napredni.js

// Metoda za posredovanje statične vsebine (datoteke iz mape public)
function posredujStaticnoVsebino(odgovor, potDoDatoteke) {
  fs.access(potDoDatoteke, (napaka) => {
    if (!napaka) {
      fs.readFile(potDoDatoteke, (napaka, datotekaVsebina) => {
        if (napaka) {
          posredujNapako(odgovor, 500);
        } else {
          posredujDatoteko(odgovor, potDoDatoteke, datotekaVsebina);
        }
      });
    } else {
      posredujNapako(odgovor, 404);
    }
  });
}

8.3.7 Pripravimo metodo za obvladovanje napak

Če je odgovornost za napako na strani odjemalca, vrnemo napako 404, saj gre za zahtevo po neobstoječem viru. V primeru napake strežnika pa uporabniku vrnemo napako 500.

streznik_napredni.js

// Obvladovanje napak
function posredujNapako(odgovor, tip) {
  odgovor.writeHead(tip, { "Content-Type": "text/plain" });
  if (tip == 404) {
    odgovor.end("Napaka 404: Vira ni mogoče najti.");
  } else if (tip == 500) {
    odgovor.end("Napaka 500: Prišlo je do napake strežnika.");
  } else {
    odgovor.end("Napaka " + tip + ": Neka druga napaka");
  }
}

8.3.8 Pripravimo tudi metodo, ki vrne datoteko odjemalcu

streznik_napredni.js

// Metoda, ki vrne datoteko in nastavi tip datoteke,
// da jo brskalnik zna ustrezno prikazati
function posredujDatoteko(odgovor, datotekaPot, datotekaVsebina) {
  odgovor.writeHead(200, {

}

Spremembe v repozitoriju od tega koraka naprej so na voljo na uveljavitvi s sporočilom Dodano predpomnjenje.

Najprej v datoteki streznik_napredni.js opredelimo objekt predpomnilnik, kamor bomo shranjevali statične datoteke.

const path = require("path");
let predpomnilnik = {};
// Kreiramo objekt strežnika in mu nastavimo osnovne parametre

Nato posodobimo metodo posredujStaticnoVsebino za posredovanje statične vsebine datotek.

streznik_napredni.js

// Metoda za posredovanje statične vsebine (datoteke iz mape public)
function posredujStaticnoVsebino(odgovor, potDoDatoteke) {
  // Preverjanje, ali je datoteka v predpomnilniku
  if (predpomnilnik[potDoDatoteke]) {
    posredujDatoteko(odgovor, potDoDatoteke, predpomnilnik[potDoDatoteke]);
  } else {
    fs.access(potDoDatoteke, (napaka) => {
      if (!napaka) {
        fs.readFile(potDoDatoteke, (napaka, datotekaVsebina) => {
          if (napaka) {
            posredujNapako(odgovor, 500);
          } else {
            // Shranjevanje vsebine nove datoteke v predpomnilnik
            predpomnilnik[potDoDatoteke] = datotekaVsebina;
            posredujDatoteko(odgovor, potDoDatoteke, datotekaVsebina);
          }
        });
      } else {
        posredujNapako(odgovor, 404);
      }
    });
  }
}

8.5 Strežniku dodamo dinamične strani

Spremembe v repozitoriju od tega koraka naprej so na voljo na uveljavitvi s sporočilom Dinamična stran s pisanjem v datoteko.

V datoteko želimo shraniti vse komentarje, ki jih uporabniki pustijo na naši strani.

8.5.1 Popravimo nastavitev strežnika

Strežniku moramo sporočiti, kdaj naj se odzove na poslane podatke.

streznik_napredni.js

// Kreiramo objekt strežnika in mu nastavimo osnovne parametre
const streznik = http.createServer((zahteva, odgovor) => {
  let potDoDatoteke = false;
  if (zahteva.url == "/") {
    potDoDatoteke = "./public/index.html";
  } else if (zahteva.url == "/posredujSporocilo") {
    obdelajSporocilo(zahteva, odgovor);
  } else {
    potDoDatoteke = "./public" + zahteva.url;
  }
  // Statično vsebino postrežemo le takrat, ko gre za zahtevo takšne vrste
  if (potDoDatoteke)
    posredujStaticnoVsebino(odgovor, potDoDatoteke);
});

8.5.2 Implementiramo metodo za obdelavo sporočila

Uporabili bomo knjižnico formidable, ki nam omogoča enostavno razčlenjevanje posredovanih podatkov z metodo parse nad objektom formidable.IncomingForm preko spletnega obrazca. V primeru napake na strani strežnika o tem obvestimo tudi uporabnika. Če so podatki pravilno posredovani, jih shranimo v datoteko sporocila.txt in se zahvalimo uporabniku.

streznik_napredni.js

// Metoda, ki obdela sporočilo na strežniku
function obdelajSporocilo(zahteva, odgovor) {
  let obrazec = new formidable.IncomingForm();
  obrazec.parse(zahteva, (napaka, vrednosti) => {
    if (napaka) {
      posredujNapako(odgovor, 404);
    } else {
      let sporociloUporabnika =
        new Date().toJSON() +
        "|" +
        vrednosti.oseba +
        "|" +
        vrednosti.komentar +
        "|" +
        vrednosti.strah +
        "\n";
      fs.appendFile("./sporocila.txt", sporociloUporabnika, (napaka) => {
        if (napaka) {
          posredujNapako(odgovor, 500);
        } else {
          posredujZahvaloZaPosredovanKomentar(odgovor);
        }
      });
    }
  });
}
