Primerček v zbirnem jeziku za x86-64 (le informativno)

Zahteve zaključka 
;----------------------------------------------------------------------------
;   Izračuna C=A+B in rezultat izpiše na standardni izhod (za 64-bitni Linux)
;
;   nasm -f elf64 p1.asm && ld -o p1 p1.o && ./p1
; ---------------------------------------------------------------------------
		global    _start
		section   .data
A:		db 	4
B:		db	5
C:		db	0, 10	        ; 10: ASCII znak za novo vrstico ('\n')
		section   .text
_start:   
		mov	al, [A]	        ; al <- M[A]
		add 	al, [B]		; al <- al + M[B]
		add 	al, 48		; al <- al + 48 ('hack' za pretvorbo cifre v ASCII znak)
		mov 	[C], al		; M[C] <- al
		mov       rax, 1	; sistemski klic write (v jeziku C: write(fd,buf,count))
		mov       rdi, 1	; datotečni opisnik (fd) za standardni izhod
		mov       rsi, C	; naslov niza (buf)
		mov       rdx, 2	; st. bajtov (count)
		syscall			; sw. prekinitev (interrupt) (prej int 0x80)
		
		mov       rax, 60	; sistemski klic exit
		xor       rdi, rdi	; izhodna vrednost 0 (v jeziku C: exit(0) )
		syscall
